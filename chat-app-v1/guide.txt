# Để gọi API đến backend, thì backend phải thêm @CrossOrigin, cho chạy đồng thời BE và FE
@RestController
@CrossOrigin(origins = "http://localhost:3000")

# Ở FE dùng axios gọi API
useEffect(() => {
        axios
            .get("http://localhost:8080/list")
            .then(res => {
                setState({users: res.data}); 
            })
            .catch(err => {
                throw err;
            })
        },[]);
